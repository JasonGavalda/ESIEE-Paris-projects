004/**
005 * This class is part of the "Space Hermes" application. 
006 * "Space Hermes" is a very simple, text based adventure game.  
007 * 
008 * This class holds an enumeration table of all command words known to the game.
009 * It is used to recognise commands as they are typed in.
010 *
011 * @author  Michael Kolling and David J. Barnes + D.Bureau + J.Gavalda
012 * @version 2008.03.30 + 2013.09.15 + 2019.10.29
013 */
014public class CommandWords
015{
016    // tableau constant qui contient tous les mots de commande valides
017    private static final String[] sValidCommands = {
018        "go", "quit", "help", "look", "use", "back", "test", "take", "drop", "Inventory", "charge", "warp", "alea"
019    };
020
021    /**
022     * Default constructor.
023     */
024    public CommandWords()
025    {
026        // rien a faire pour le moment...
027    } // CommandWords()
028
029    /**
030     * Checks if a String is part of the validCommands.
031     * @param pString The String to test
032     * @return True if pString is a valid command, neither returns false.
033     */
034    public boolean isCommand( final String pString )
035    {
036        // pour chacune des commandes valides (du tableau constant)
037        for ( int i=0; i<sValidCommands.length; i++ ) {
038            // si elle est egale a pString
039            if ( sValidCommands[i].equals( pString ) )
040                return true;
041        } // for
042        // si nous arrivons la, c'est que la commande
043        // n'a pas ete trouvee dans le tableau
044        return false;
045    } // isCommand()
046    
047    /**
048     * Gets all the valid commands
049     * @return A String containing all the valid commands.
050     */
051    public String getCommandList()
052    {   
053        String vString="";
054        for(String vCommand : sValidCommands)
055        {
056            vString = vString + vCommand + " ";
057        }
058        return vString;
059    } // getCommandList()
060} // CommandWords
