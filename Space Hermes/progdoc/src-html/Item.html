002/**
003 * Class of the Items of the game.
004 * Each Item is defined by its name, its description and its weight.
005 * @author Jason GAVALDA
006 * @version 22/11/2019
007 */
008public class Item
009{
010    private String aName;
011    private String aDescription;
012    private double aWeight;
013    
014    /**
015     * Constructor creating an item with its name, its description and its weight.
016     * @param pName The name of the item.
017     * @param pDescription The description of the Item.
018     * @param pWeight The weight of the Item.
019     */
020    public Item (final String pName, final String pDescription, final double pWeight)  {
021        this.aName = pName;
022        this.aDescription = pDescription;
023        this.aWeight = pWeight;
024    } // Item()
025    
026    /**
027     * Accessor which gets the Name of an Item.
028     * @return The name of an Item.
029     */
030    public String getItemName()
031    {
032        return this.aName;
033    } // getItemName()
034    
035    /**
036     * Accessor which gets the description of an Item.
037     * @return The description of an Item.
038     */
039    public String getItem_Description() {
040        return this.aDescription;
041    } // getItem_Description()
042    
043    /**
044     * Accessor which gets the weight of an Item.
045     * @return The weight of an Item.
046     */
047    public double getWeight()  {
048        return this.aWeight;
049    } // getWeight()
050} // Item
