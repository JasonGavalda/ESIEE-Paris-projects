018/**
019 * This class implements a simple graphical user interface with a text entry
020 * area, a text output area and an optional image.
021 * 
022 * @author Michael Kolling + Jason Gavalda
023 * @version 1.0 (Jan 2003) DB edited (2019) + 8/01/2020
024 */
025public class UserInterface implements ActionListener
026{
027    private GameEngine aEngine;
028    private JFrame     aMyFrame;
029    private JTextField aEntryField;
030    private JTextArea  aLog;
031    private JLabel     aImage;
032    private JButton    aButton;
033    private JButton    aButton1;
034    private JButton    aButton2;
035    private JButton    aButton3;
036    private JButton    aButton4;
037    private JButton    aButton5;
038    private JButton    aButton6;
039    private JButton    aButton7;
040    private JButton    aButton8;
041    private JButton    aButton9;
042    private JButton    aButton10;
043    private JButton    aButton11;
044
045    /**
046     * Construct a UserInterface. As a parameter, a Game Engine
047     * (an object processing and executing the game commands) is
048     * needed.
049     * 
050     * @param pGameEngine  The GameEngine object implementing the game logic.
051     */
052    public UserInterface( final GameEngine pGameEngine )
053    {
054        this.aEngine = pGameEngine;
055        this.createGUI();
056    } // UserInterface(.)
057
058    /**
059     * Print out some text into the text area.
060     * @param pText The Text to be displayed.
061     */
062    public void print( final String pText )
063    {
064        this.aLog.append( pText );
065        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
066    } // print(.)
067
068    /**
069     * Print out some text into the text area, followed by a line break.
070     * @param pText The text to be displayed.
071     */
072    public void println( final String pText )
073    {
074        this.print( pText + "\n" );
075    } // println(.)
076
077    /**
078     * Show an image file in the interface.
079     * @param pImageName The Name of the image to be displayed.
080     */
081    public void showImage( final String pImageName )
082    {
083        String vImagePath = "" + pImageName; // to change the directory
084        URL vImageURL = this.getClass().getClassLoader().getResource( vImagePath );
085        if ( vImageURL == null )
086            System.out.println( "Image not found : " + vImagePath );
087        else {
088            ImageIcon vIcon = new ImageIcon( vImageURL );
089            this.aImage.setIcon( vIcon );
090            this.aMyFrame.pack();
091        }
092    } // showImage(.)
093
094    /**
095     * Enable or disable input in the input field.
096     * @param pOnOff A boolean determinating the authorization of the field to be displayed.
097     */
098    public void enable( final boolean pOnOff )
099    {
100        this.aEntryField.setEditable( pOnOff ); // enable/disable
101        
102        if ( ! pOnOff ) { // disable
103            this.aEntryField.getCaret().setBlinkRate( 0 ); // cursor won't blink
104            this.aEntryField.removeActionListener( this ); // won't react to entry
105            this.aButton.removeActionListener ( this );
106            this.aButton1.removeActionListener ( this );
107            this.aButton2.removeActionListener ( this );
108            this.aButton3.removeActionListener ( this );
109            this.aButton3.removeActionListener ( this );
110            this.aButton4.removeActionListener ( this );
111            this.aButton5.removeActionListener ( this );
112            this.aButton6.removeActionListener ( this );
113            this.aButton7.removeActionListener ( this );
114            this.aButton8.removeActionListener ( this );
115            this.aButton9.removeActionListener ( this );
116            this.aButton10.removeActionListener ( this );
117            this.aButton11.removeActionListener ( this );            
118        }
119    } // enable(.)
120
121    /**
122     * Set up graphical user interface.
123     */
124    private void createGUI()
125    {
126        this.aMyFrame = new JFrame( "Hermes" );
127        this.aEntryField = new JTextField( 34 );
128
129        this.aLog = new JTextArea();
130        this.aLog.setEditable( false );
131        JScrollPane vListScroller = new JScrollPane( this.aLog );
132        vListScroller.setPreferredSize( new Dimension(800, 300) );
133        vListScroller.setMinimumSize( new Dimension(100,100) );
134
135        JPanel vPanel = new JPanel();
136        JPanel vPanel1 = new JPanel();
137        JPanel vPanel2 = new JPanel();
138        
139        this.aImage = new JLabel();
140        
141        this.aButton = new JButton( "warp" );
142        this.aButton1 = new JButton( "charge" );
143        this.aButton2 = new JButton( "Inventory");
144        this.aButton3 = new JButton( "back" );
145        this.aButton4 = new JButton( "look" );
146        this.aButton5 = new JButton( "go north" );
147        this.aButton6 = new JButton( "go east" );
148        this.aButton7 = new JButton( "go south" );
149        this.aButton8 = new JButton( "go west" );
150        this.aButton9 = new JButton( "go up" );
151        this.aButton10 = new JButton( "go down" );
152        this.aButton11 = new JButton( "quit" );
153        
154        vPanel.setLayout( new BorderLayout() ); // ==> only five places
155        vPanel1.setLayout( new GridLayout(2,3) );
156        vPanel2.setLayout( new GridLayout(2,3) );
157        
158        vPanel1.add(this.aButton3);
159        vPanel1.add(this.aButton4);
160        vPanel1.add(this.aButton2);
161        vPanel1.add(this.aButton);
162        vPanel1.add(this.aButton1);
163        vPanel1.add(this.aButton11);
164        
165        vPanel2.add(this.aButton10);
166        vPanel2.add(this.aButton5);
167        vPanel2.add(this.aButton9);
168        vPanel2.add(this.aButton8);
169        vPanel2.add(this.aButton7);
170        vPanel2.add(this.aButton6);
171        
172        vPanel.add( this.aImage, BorderLayout.NORTH );
173        this.aImage.setHorizontalAlignment(javax.swing.SwingConstants.CENTER);
174        vPanel.add( vListScroller, BorderLayout.CENTER );
175        vPanel.add( this.aEntryField, BorderLayout.SOUTH );

181        // add some event listeners to some components
182        this.aEntryField.addActionListener( this );
183        this.aButton.addActionListener ( this );
184        this.aButton1.addActionListener ( this );
185        this.aButton2.addActionListener ( this );
186        this.aButton3.addActionListener ( this );
187        this.aButton4.addActionListener ( this );
188        this.aButton5.addActionListener ( this );
189        this.aButton6.addActionListener ( this );
190        this.aButton7.addActionListener ( this );
191        this.aButton8.addActionListener ( this );
192        this.aButton9.addActionListener ( this );
193        this.aButton10.addActionListener ( this );
194        this.aButton11.addActionListener ( this );
195        
196        // to end program when window is closed
197        this.aMyFrame.addWindowListener( new WindowAdapter() {
198            public void windowClosing(WindowEvent e) { System.exit(0); }
199        } );
200
201        this.aMyFrame.pack();
202        this.aMyFrame.setVisible( true );
203        this.aEntryField.requestFocus();
204    } // createGUI()
205
206    /**
207     * Actionlistener interface for entry textfield.
208     * @param pE The action to be performed.
209     */
210    public void actionPerformed( final ActionEvent pE ) 
211    {
212        Object vSource = pE.getSource();
213        
214        if (vSource == this.aButton | vSource == this.aButton1 | vSource == this.aButton2 | vSource == this.aButton3 | vSource == this.aButton4 | vSource == this.aButton5 | vSource == this.aButton6 | vSource == this.aButton7 | vSource == this.aButton8 | vSource == this.aButton9 | vSource == this.aButton10 | vSource == this.aButton11)
215            this.aEngine.interpretCommand(pE.getActionCommand());
216        else
217        this.processCommand();
218    } // actionPerformed(.)
219
220    /**
221     * A command has been entered. Read the command and do whatever is 
222     * necessary to process it.
223     */
224    private void processCommand()
225    {
226        String vInput = this.aEntryField.getText();
227        this.aEntryField.setText( "" );
228
229        this.aEngine.interpretCommand( vInput );
230    } // processCommand()
231} // UserInterface
