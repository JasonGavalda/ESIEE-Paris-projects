002/**
003 * A precise type of Room used to teleport a player 
004 * to another random selected Room.
005 *
006 * @author Jason GAVALDA
007 * @version 12.01.2020
008 */
009public class TransporterRoom extends Room
010{
011    private RoomRandomizer aRoomRandomizer;
012    
013    /**
014     * Constructor which creates a TransporterRoom using the Room constructor.
015     * @param pDescription The Description of the TransporterRoom
016     * @param pImage The image of the TransporterRoom.
017     */
018    public TransporterRoom (final String pDescription , final String pImage)
019    {
020        super(pDescription ,pImage);
021        this.aRoomRandomizer = new RoomRandomizer();
022    } // Transporter_Room
023    
024    /**
025     * Accessor which gets the RoomRandomizer of the TransporterRoom.
026     * @return The RoomRandomizer of the TransporterRoom.
027     */
028    public RoomRandomizer getRoomRandomizer()
029    {
030        return this.aRoomRandomizer;
031    }
032    
033    /**
034     * Overrides the previous getExit() method from Room to get a random exit
035     * for the current TransporterRoom rather than a normal exit.
036     * @param pDirection The Direction the player wants to moves in, 
037     * useless in this method but used to be interpretated correctly by goRoom() in GameEngine.
038     * @return A random exit Room among those attached to the TransporterRoom.
039     */
040    @Override
041    public Room getExit(final String pDirection)
042    {
043        return this.aRoomRandomizer.findRandomRoom();
044    } // getExit()
045    
046    /**
047     * Adds a Room attached to the TransporterRoom.
048     * @param pRoom The Room to add to the list of attached Rooms in this TransporterRoom.
049     */
050    public void addExit(final Room pRoom)
051    {
052        this.aRoomRandomizer.addRoom(pRoom);
053    } // addExit()
054    
055    /**
056     * Removes a Room attached to the TransporterRoom.
057     * @param pRoom The Room to add to the list of attached Rooms in this TransporterRoom.
058     */
059    public void removeExit(final Room pRoom)
060    {
061        if (this.aRoomRandomizer.roomCheck(pRoom))
062            this.aRoomRandomizer.removeRoom(pRoom);
063    } // removeExit()
064    
065} // TransporterRoom
