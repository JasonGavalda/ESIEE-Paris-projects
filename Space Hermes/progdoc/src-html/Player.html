001import java.util.Stack;
002/**
003 * This class describes the Players objects.
004 * Each player is defined by his name,
005 * his inventory,
006 * the maximum weight his vessel can carry,
007 * the weight of his inventory,
008 * the current Room he is placed and
009 * the previous Rooms he visited,
010 *
011 * @author Jason GAVALDA
012 * @version 10/12/2019
013 */
014public class Player
015{
016    private Room aCurrentRoom;
017    private Stack aPreviousRoom;
018    private String aName;
019    private ItemList aInventory;
020    private double aMaxWeight;
021    private double aInventoryWeight;
022    private int aFuel;
023    private Stack aStarGate;
024    
025    /**
026     * Constructor creating a player with a name and a starting Room chosen.
027     * His inventory is empty when created.
028     * One player can carry a maximum weight of 30.
029     * @param pName The name of the player.
030     * @param pCurrentRoom The starting Room of the player.
031     */
032    public Player (final String pName, final Room pCurrentRoom)
033    {
034        this.aName = pName;
035        this.aPreviousRoom = new Stack();
036        this.aCurrentRoom = pCurrentRoom;   
037        this.aInventory = new ItemList();
038        this.aMaxWeight = 30;
039        this.aInventoryWeight = 0;
040        this.aFuel = 5;
041        this.aStarGate = new Stack();
042    }
043    
044    /**
045     * Accessor to get the current Room of a player.
046     * @return The current Room of a player.
047     */
048    public Room getCurrentRoom()
049    {
050        return this.aCurrentRoom;
051    } // getCurrentRoom()
052    
053    /**
054     * Accessor to get the previous Rooms the player went in.
055     * @return A Stack of the previous Room encountered by the player.
056     */
057    public Stack getPreviousRooms()
058    {
059        return this.aPreviousRoom;
060    } // getPreviousRooms()
061    
062    /**
063     * Get the aPreviousRoom Stack empty
064     */
065    public void getStackEmpty()
066    {
067        while (!(this.aPreviousRoom.empty()))
068        {
069            this.aPreviousRoom.pop();
070        }
071    } // getStackEmpty
072    
073    /**
074     * Accessor to get the maximum weight a player can carry.
075     * @return The maximum weight a player can carry on his vessel.
076     */
077    public double getMaxWeight()
078    {
079        return this.aMaxWeight;
080    } // getMaxWeight()
081    
082    /**
083     * Accessor to get the weight a player his currently carrying in his vessel.
084     * @return The weight a player is carrying in his vessel.
085     */
086    public double getInventoryWeight()
087    {
088        return this.aInventoryWeight;
089    } // getInventoryWeight()
090    
091    /**
092     * Accessor which gets the remaining fuel of the player.
093     * @return The remaining fuel of the player.
094     */
095    public int getFuel()
096    {
097        return this.aFuel;
098    } // getFuel()
099    
100    /**
101     * Accessor which gets the aStarGate stack.
102     * @return The aStarGate stack.
103     */
104    public Stack getStarGate()
105    {
106        return this.aStarGate;
107    } // getStarGate()
108    
109    /**
110     * Modifier which modifies the maximum weight the player can carry.
111     * @param pMaxWeight The new max weight a player is able to carry.
112     */
113    public void setMaxWeight(final double pMaxWeight)
114    {
115        this.aMaxWeight = pMaxWeight;
116    } // setMaxWeight()
117    
118    /**
119     * Modifier which modifies the remaining fuel of the player.
120     * @param pFuel The new amount of fuel the player will have.
121     */
122    public void setFuel(final int pFuel)
123    {
124        this.aFuel = pFuel;
125    } // setFuel()
126    
127    /**
128     * The actual part of the "go" method where the player actually moves one Room to another.
129     * @param pNextRoom The next Room where the player moves in.
130     */
131    public void move(final Room pNextRoom)
132    {
133        this.aPreviousRoom.push(this.aCurrentRoom);
134        this.aCurrentRoom = pNextRoom;
135        this.aFuel -= 1;
136    } // move()
137    
138    /**
139     * The actual part of the "back" method where the player actually moves back to the previous Room he went in.
140     */
141    public void moveBack()
142    {
143        this.aCurrentRoom =(Room) this.aPreviousRoom.pop();
144        this.aFuel -= 1;
145    } // moveBack()
146    
147    /**
148     * This method checks if the stack aPreviousRoom is empty or not.
149     * @return Whether the aPreviousRoom is empty or not.
150     */
151    public boolean isBackEmpty()
152    {
153        return this.aPreviousRoom.empty();
154    }
155    
156    /**
157     * A command that allows a player to take a specified item in a Room.
158     * @param pItemName The name of the item to be taken.
159     */
160    public void take (final String pItemName)
161    {
162        Item vItem = this.aCurrentRoom.getRoomItem(pItemName);
163        this.aCurrentRoom.removeRoomItem(pItemName);
164        this.aInventory.addItem(pItemName, vItem);
165        this.aInventoryWeight = this.aInventoryWeight + vItem.getWeight();
166    } // take()
167    
168    /**
169     * A command that allows a player to drop an item of his inventory in the Room he currently is.
170     * @param pItemName The name of the item to be dropped.
171     */
172    public void drop (final String pItemName)
173    {
174        Item vItem = this.aInventory.getItem(pItemName);
175        this.aInventory.removeItem(pItemName);
176        this.aCurrentRoom.setItem(pItemName, vItem);
177        this.aInventoryWeight = this.aInventoryWeight - vItem.getWeight();
178    } // drop()
179    
180    /**
181     * A command that allows the player to charge the StarGate with the Room he is in.
182     */
183    public void charge()
184    {
185        if (!(isStarGateEmpty()))
186            this.aStarGate.pop();
187        this.aStarGate.push(this.aCurrentRoom);
188    } // charge()
189    
190    /**
191     * A command that allows the player to teleport in the Room stocked in the StarGate.
192     */
193    public void warp()
194    {
195        Room vNextRoom = (Room) this.aStarGate.pop();
196        this.move(vNextRoom);
197    } // warp()
198    
199    /**
200     * Checks whether the StarGate is empty or not
201     * @return True if the StarGate is empty or false either.
202     */
203    public boolean isStarGateEmpty()
204    {
205        return (this.aStarGate.empty());
206    } // isStarGateEmpty()
207    
208    /**
209     * Accessor which gets the Inventory as an ItemList.
210     * @return The Inventory as an ItemList.
211     */
212    public ItemList getInventory ()
213    {
214        return this.aInventory;
215    } // getInventory ()
216    
217    /**
218     * @return A String showing the Items the player's vessel is carrying.
219     */
220    public String getPlayerInventory()
221    {
222        return "In your hold, "+this.aInventory.getItemString();
223    } // getPlayerInventory()
224} // Player
